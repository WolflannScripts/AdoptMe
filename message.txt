function ScanForRegister()
    u63 = ''
    u64 = ''

    local v251 = workspace.HouseInteriors:WaitForChild('furniture'):GetChildren()
    local v252, v253, v254 = pairs(v251)

    while true do
        local v255

        v254, v255 = v252(v253, v254)

        if v254 == nil then
            break
        end

        task.wait()

        if not (v255:IsA('Folder') and v255:FindFirstChildWhichIsA('Model')) then
            Notification('You must be in a house to scan')
            _Scaninfonone:Set('Scan info: failed', 'circle-x', nil, false)

            return
        end
        if string.find(v255.Name, game.Players.LocalPlayer.Name) then
            Notification("you can't scan in your own house")
            _Scaninfonone:Set('Scan info: failed', 'circle-x', nil, false)

            return
        end
        if v255:FindFirstChild('CashRegister') then
            u63 = u13.getLastSegment(v255.Name)
            u64 = u13.GetHouseInterior()

            if string.find(u63, 'f-') then
                _Scaninfonone:Set('Scan info: successful', 'circle-check', nil, false)
            else
                Notification('something went wrong with the scanning')
                _Scaninfonone:Set('Scan info: failed', 'circle-x', nil, false)

                u63 = ''
                u64 = ''
            end

            return
        end
    end

    _Scaninfonone:Set('Scan info: failed', 'circle-x', nil, false)
    Notification('cash register was not found')
end
